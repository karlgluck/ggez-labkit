using System;
using UnityEngine;
using System.Collections;
using System.Collections.Generic;
using UnityEngine.Events;

namespace GGEZ
{




//----------------------------------------------------------------------
[Serializable]
public class UnityEventWith%CAPITALIZEDTYPE% : UnityEvent<%TYPE%>
{
}



//----------------------------------------------------------------------
//----------------------------------------------------------------------
[
AddComponentMenu ("GGEZ/Game Register Listener/%TYPE%")
]
public class GameRegisterListener : MonoBehaviour
{



#region Serialized
[Header ("Serialized")]
[SerializeField] private GameRegister gameRegisterIn;
[SerializeField] private UnityEventWith%CAPITALIZEDTYPE% unityEventOut;
#endregion




void OnEnable ()
    {
    if (this.gameRegisterIn != null)
        {
        this.gameRegisterIn.RegisterListener (this);
        }
    }




void OnDisable ()
    {
    if (this.gameRegisterIn != null)
        {
        this.gameRegisterIn.UnregisterListener (this);
        }
    }




public void OnDidChange (%TYPE% newValue)
    {
    this.unityEventOut.Invoke (newValue);
    }




//----------------------------------------------------------------------
// Handle the Unity Editor changing gameEgameRegisterIn in the inspector
//----------------------------------------------------------------------
#if UNITY_EDITOR
#region Editor Runtime
[Header ("Editor Runtime")]
private GameRegister lastGameRegisterIn;




void OnValidate ()
    {
    this.validateGameRegisterIn ();
    }




private void validateGameRegisterIn ()
    {
    if (object.ReferenceEquals (this.gameRegisterIn, this.lastGameRegisterIn))
        {
        return;
        }
    if (this.lastGameRegisterIn != null)
        {
        this.lastGameRegisterIn.UnregisterListener (this);
        }
    if (this.gameRegisterIn != null)
        {
        this.gameRegisterIn.RegisterListener (this);
        }
    this.lastGameRegisterIn = this.gameRegisterIn;
    }

#endregion
#endif






}


}
